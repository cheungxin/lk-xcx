## 模块一：练习记录页（record）

### 1.1 当前状态

`record.js` 使用固定 Mock 数据 `MOCK_RECORDS`，`handleRecordTap` 仅弹出 toast。

### 1.2 目标功能

1. 从 `storage.getPracticeRecords()` 读取真实练习记录（由答题页交卷时写入）
2. 点击记录条目，根据练习类型跳转到对应详情：
   - 类型为「估分」/「模拟刷题」→ 跳转考试报告页
   - 类型为「专项练习」/「每日一练」/「快练」→ 跳转解析页
3. 记录支持按类型筛选（全部 / 快练 / 专项练习 / 估分 / 每日一练 / 模拟刷题）
4. 记录按日期分组，按日期倒序排列

### 1.3 修改 `record.js`

#### 完整改动方案：

```javascript
import router from '../../../router/index'
import storage from '../../../utils/storage'
const tool = require('../../../toolbox/tool')

Page({
  data: {
    navBarHeight: 88,
    records: [],
    activeTab: 0,
    filteredRecords: [],
    filterTabs: ['全部', '快练', '专项练习', '估分', '每日一练', '模拟刷题'],
  },

  onLoad() {
    const systemInfo = wx.getSystemInfoSync()
    const statusBarHeight = systemInfo.statusBarHeight || 0
    this.setData({ navBarHeight: statusBarHeight + 44 })
    this.loadRecords()
  },

  onShow() {
    // 每次回到页面时刷新（答题后回来能看到新记录）
    this.loadRecords()
  },

  loadRecords() {
    const rawRecords = storage.getPracticeRecords()
    // 按日期分组
    const grouped = this.groupRecordsByDate(rawRecords)
    this.setData({ 
      records: grouped,
      filteredRecords: grouped,
    })
  },

  groupRecordsByDate(rawList) {
    const map = {}
    const today = new Date().toISOString().split('T')[0]
    const yesterday = new Date(Date.now() - 86400000).toISOString().split('T')[0]

    rawList.forEach(record => {
      const date = record.date || today
      if (!map[date]) {
        map[date] = {
          date,
          weekday: date === today ? '今天' : date === yesterday ? '昨天' : this.getWeekday(date),
          records: [],
          summary: { total: 0, correct: 0, time: '' },
        }
      }
      map[date].records.push(record)
      map[date].summary.total += record.questions || 0
      map[date].summary.correct += record.correct || 0
    })

    // 转为数组，按日期倒序
    return Object.keys(map)
      .sort((a, b) => b.localeCompare(a))
      .map(date => map[date])
  },

  getWeekday(dateStr) {
    const days = ['周日', '周一', '周二', '周三', '周四', '周五', '周六']
    return days[new Date(dateStr).getDay()]
  },

  onTabChange(e) {
    const index = e.detail.value
    this.setData({ activeTab: index })
    const tab = this.data.filterTabs[index]
    
    const allRecords = storage.getPracticeRecords()
    const filtered = tab === '全部'
      ? allRecords
      : allRecords.filter(r => r.type === tab)
    
    this.setData({ 
      filteredRecords: this.groupRecordsByDate(filtered) 
    })
  },

  handleRecordTap(e) {
    const { id, type } = e.currentTarget.dataset
    
    // 根据类型决定跳转目标
    if (type === '估分' || type === '模拟刷题') {
      // 考场模式 → 跳转考试报告页（需要先从 storage 找到对应结果）
      const result = storage.getExamResultById(id)
      if (result) {
        storage.setExamResult(result)   // 设为当前结果，报告页从此读取
        router.push('EXAM_REPORT')
      } else {
        tool.toast('暂无详细报告数据')
      }
    } else {
      // 专项练习 / 每日一练 / 快练 → 跳转解析页
      const result = storage.getPracticeResultById(id)
      if (result) {
        storage.setPracticeResult(result)  // 设为当前结果，解析页从此读取
        router.push('ANALYSIS', { practiceType: type === '每日一练' ? 'daily' : 'practice' })
      } else {
        tool.toast('暂无解析数据')
      }
    }
  },

  handleBack() {
    wx.navigateBack({ delta: 1 })
  },
})
```

### 2.4 storage.js 新增错题本方法

```javascript
// 添加错题（自动去重：同一题不重复添加，但更新最近出错记录）
addWrongQuestion(question) {
  const list = wx.getStorageSync('wrong_questions') || []
  const existIndex = list.findIndex(q => q.id === question.id)
  if (existIndex > -1) {
    // 已有该题，更新出错次数和时间
    list[existIndex].errorCount = (list[existIndex].errorCount || 1) + 1
    list[existIndex].lastErrorAt = question.addedAt
    list[existIndex].mastered = false  // 重新出错，重置掌握状态
  } else {
    list.unshift({ ...question, errorCount: 1 })
  }
  wx.setStorageSync('wrong_questions', list)
},

getWrongQuestions(filters = {}) {
  const list = wx.getStorageSync('wrong_questions') || []
  let result = [...list]
  
  if (filters.category) {
    result = result.filter(q => q.category === filters.category)
  }
  if (filters.mastered !== undefined) {
    result = result.filter(q => !!q.mastered === filters.mastered)
  }
  return result
},

markWrongQuestionMastered(questionId) {
  const list = wx.getStorageSync('wrong_questions') || []
  const updated = list.map(q => q.id === questionId ? { ...q, mastered: true } : q)
  wx.setStorageSync('wrong_questions', updated)
},

removeWrongQuestion(questionId) {
  const list = wx.getStorageSync('wrong_questions') || []
  wx.setStorageSync('wrong_questions', list.filter(q => q.id !== questionId))
},

clearWrongQuestions() {
  wx.removeStorageSync('wrong_questions')
},
```

### 2.5 修改 wrong.js 读取真实错题数据

```javascript
loadWrongQuestions() {
  const { categoryValue, statusValue, sortValue } = this.data
  
  const filters = {}
  if (statusValue === 1) filters.mastered = false
  if (statusValue === 2) filters.mastered = true
  
  const wrongQuestions = storage.getWrongQuestions(filters)
  
  // 按知识点分类分组
  const categoryMap = {}
  wrongQuestions.forEach(item => {
    const category = item.category || '其他'
    if (!categoryMap[category]) {
      categoryMap[category] = { name: category, count: 0, items: [], expanded: false }
    }
    categoryMap[category].count++
    categoryMap[category].items.push(item)
  })

  const categories = Object.values(categoryMap)
  
  // 按试卷分组（供试卷 Tab 使用）
  const paperMap = {}
  wrongQuestions.forEach(item => {
    const paper = item.source || '其他来源'
    if (!paperMap[paper]) {
      paperMap[paper] = { name: paper, count: 0, items: [] }
    }
    paperMap[paper].count++
    paperMap[paper].items.push(item)
  })

  this.setData({
    knowledgeCategories: categories,
    paperList: Object.values(paperMap),
    totalCount: wrongQuestions.length,
  })
},
```

## 模块四：学习统计（stats）

### 4.1 当前状态

`stats.js` 全部使用 Mock 数据，纯 UI 展示。

### 4.2 目标功能（分两步实现）

**第一步**（基础）：保留现有 UI，将 Mock 数据替换为从 storage 动态计算的真实数据。  
**第二步**（进阶）：按选择的时间维度（本周/本月/全部）动态切换数据。

### 4.3 从 storage 计算统计数据

```javascript
// 在 stats.js 的 onLoad/onShow 中：
loadStats() {
  const records = storage.getPracticeRecords()  // 全部练习记录
  
  // 1. 概览统计
  const totalQuestions = records.reduce((sum, r) => sum + (r.questions || 0), 0)
  const totalCorrect = records.reduce((sum, r) => sum + (r.correct || 0), 0)
  const correctRate = totalQuestions > 0 ? Math.round((totalCorrect / totalQuestions) * 100) : 0
  
  // 今日数据
  const today = new Date().toISOString().split('T')[0]
  const todayRecords = records.filter(r => r.date === today)
  const todayQuestions = todayRecords.reduce((sum, r) => sum + (r.questions || 0), 0)
  
  // 连续打卡天数（从 daily_records 计算）
  const dailyRecords = storage.getDailyRecords()
  const continueDays = this.calcContinueDays(dailyRecords)
  
  // 2. 近7天数据
  const dailyData = this.calcWeeklyData(records)
  
  // 3. 薄弱环节（从错题本计算）
  const wrongQuestions = storage.getWrongQuestions()
  const weakPoints = this.calcWeakPoints(wrongQuestions)
  
  this.setData({
    overview: { totalQuestions, correctRate, continueDays, todayQuestions },
    dailyData,
    weakPoints,
  })
},

calcContinueDays(dailyRecords) {
  let count = 0
  const today = new Date()
  for (let i = 0; i < 365; i++) {
    const date = new Date(today - i * 86400000).toISOString().split('T')[0]
    if (dailyRecords[date] && dailyRecords[date].done) {
      count++
    } else if (i > 0) {
      break  // 断了就停
    }
  }
  return count
},

calcWeeklyData(records) {
  const days = ['周日', '周一', '周二', '周三', '周四', '周五', '周六']
  const result = []
  for (let i = 6; i >= 0; i--) {
    const d = new Date(Date.now() - i * 86400000)
    const dateStr = d.toISOString().split('T')[0]
    const dayRecords = records.filter(r => r.date === dateStr)
    const count = dayRecords.reduce((s, r) => s + (r.questions || 0), 0)
    const correct = dayRecords.reduce((s, r) => s + (r.correct || 0), 0)
    result.push({ day: days[d.getDay()], count, correct })
  }
  return result
},

calcWeakPoints(wrongList) {
  const map = {}
  wrongList.filter(q => !q.mastered).forEach(q => {
    const key = q.knowledgePoint || q.category || '其他'
    if (!map[key]) map[key] = { name: key, total: 0, correct: 0 }
    map[key].total++
  })
  return Object.values(map)
    .map(item => ({ ...item, accuracy: Math.round((item.correct / item.total) * 100) }))
    .sort((a, b) => a.accuracy - b.accuracy)
    .slice(0, 5)  // 最弱的5个
},
```

### 5.2 数据写入时机汇总

```
用户答题 → practice.js handleOptionTap
  ├─ 答错 → storage.addWrongQuestion(...)

用户交卷 → practice.js submitPractice()
  ├─ storage.setPracticeResult(result)              // 供解析页

考场交卷 → exam.js submitExam()
  ├─ storage.setExamResult(examResult)              // 供报告页
  ├─ storage.appendExamResultHistory(examResult)    // 供记录页点击
  └─ storage.appendRecord({ type: '模拟刷题', ... })
```
